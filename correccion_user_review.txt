import ast
import json
import pandas as pd

# Lista para almacenar los elementos del archivo JSON
listaitems = []

# Lee el archivo JSON y agrega cada l√≠nea a la lista
with open('australian_users_items.json', 'r', encoding='utf-8') as f:
    for line in f.readlines():
        listaitems.append(ast.literal_eval(line))

# Crea un DataFrame a partir de la lista
df_user_items = pd.DataFrame(listaitems)

# Imprime el DataFrame
df_user_items.head()

# Guarda el DataFrame como un archivo JSON
df_user_items.to_json('user_items_output_ok.json', orient='records', lines=True) #, force_ascii=False, indent=2)
